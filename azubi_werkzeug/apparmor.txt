#include <tunables/global>

profile azubi_werkzeug flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Allow everything except explicit denies
  capability,
  
  # File access - allow read/write/execute everywhere
  /** rwlix,
  
  # Network access
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  
  # Only deny access to truly sensitive kernel interfaces
  deny /proc/kcore rwx,
  deny /proc/kallsyms rwx,
  deny /sys/kernel/security/** rwx,
  
  # Allow signals
  signal,
  
  # Allow ptrace (needed by some debuggers, can be removed if not needed)
  ptrace,
}
